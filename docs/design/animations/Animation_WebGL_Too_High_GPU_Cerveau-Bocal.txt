import { Canvas, useFrame } from "@react-three/fiber";
import { OrbitControls } from "@react-three/drei";
import { useRef } from "react";
import * as THREE from "three";

function BrainJar({ activity = 0 }) {
  const brainRef = useRef();
  const liquidRef = useRef();

  useFrame((state, delta) => {
    const t = state.clock.elapsedTime;

    // 🧠 Pulsation cerveau
    if (brainRef.current) {
      const pulse = 1 + Math.sin(t * 3) * 0.05 * (1 + activity);
      brainRef.current.scale.set(pulse, pulse, pulse);
      brainRef.current.material.emissiveIntensity = 0.5 + activity;
    }

    // 💧 Liquide qui bouge
    if (liquidRef.current) {
      liquidRef.current.position.y =
        -0.5 + Math.sin(t * 2) * 0.02 * (1 + activity);
    }
  });

  return (
    <group>
      {/* 🧪 VERRE */}
      <mesh>
        <cylinderGeometry args={[1, 1, 2, 64]} />
        <meshPhysicalMaterial
          transmission={1}
          roughness={0}
          thickness={0.5}
          ior={1.5}
          transparent
        />
      </mesh>

      {/* 💧 LIQUIDE */}
      <mesh ref={liquidRef} position={[0, -0.5, 0]}>
        <cylinderGeometry args={[0.95, 0.95, 1, 64]} />
        <meshStandardMaterial
          color="#00d4ff"
          transparent
          opacity={0.4}
        />
      </mesh>

      {/* 🧠 CERVEAU */}
      <mesh ref={brainRef} position={[0, 0, 0]}>
        <sphereGeometry args={[0.4, 64, 64]} />
        <meshStandardMaterial
          color="#ff6b8a"
          emissive="#ff2a6d"
          emissiveIntensity={0.5}
        />
      </mesh>
    </group>
  );
}

export default function Scene() {
  return (
    <Canvas camera={{ position: [0, 0, 4] }}>
      <ambientLight intensity={0.5} />
      <pointLight position={[2, 2, 2]} intensity={2} />

      <BrainJar activity={0.8} />

      <OrbitControls enableZoom={false} />
    </Canvas>
  );
}

🫧 5. Bulles dynamiques (important pour la vie)
function Bubbles({ count = 20 }) {
  const mesh = useRef();

  useFrame((state) => {
    mesh.current.children.forEach((b, i) => {
      b.position.y += 0.01;
      if (b.position.y > 0.5) b.position.y = -0.5;
    });
  });

  return (
    <group ref={mesh}>
      {Array.from({ length: count }).map((_, i) => (
        <mesh key={i} position={[Math.random(), Math.random(), Math.random()]}>
          <sphereGeometry args={[0.02]} />
          <meshStandardMaterial transparent opacity={0.6} />
        </mesh>
      ))}
    </group>
  );
}

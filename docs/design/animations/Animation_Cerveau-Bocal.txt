⚛️ 1. Composant React prêt à l’emploi
import { motion } from "framer-motion";

export default function BrainJar({ activity = 0, onClick }) {
  return (
    <div className="jar" onClick={onClick}>
      
      {/* 💡 Glow global */}
      <div
        className="jar-glow"
        style={{
          opacity: 0.2 + activity * 0.6,
        }}
      />

      {/* 💧 Liquide */}
      <motion.div
        className="liquid"
        animate={{
          y: [0, -5, 0],
        }}
        transition={{
          repeat: Infinity,
          duration: 3 - activity,
          ease: "easeInOut",
        }}
      />

      {/* 🫧 Bulles */}
      {[...Array(8)].map((_, i) => (
        <motion.div
          key={i}
          className="bubble"
          initial={{
            y: 60,
            x: Math.random() * 40 - 20,
            opacity: 0,
          }}
          animate={{
            y: -60,
            opacity: [0, 1, 0],
          }}
          transition={{
            duration: 2 + Math.random(),
            repeat: Infinity,
            delay: i * 0.3,
          }}
        />
      ))}

      {/* 🧠 Cerveau */}
      <motion.img
        src="/brain.png"
        className="brain"
        animate={{
          scale: [1, 1.05 + activity * 0.1, 1],
          filter: `brightness(${1 + activity})`,
        }}
        transition={{
          repeat: Infinity,
          duration: 1.5,
        }}
      />
    </div>
  );
}

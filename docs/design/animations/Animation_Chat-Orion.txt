⚛️ 4. Exemple React (plug & play)
import { motion, AnimatePresence } from "framer-motion";

export default function CatBubble({ state, message }) {
  return (
    <div className="cat-container">
      
      {/* TON CHAT */}
      <img src="/cat.png" className="cat" />

      <AnimatePresence>
        {state === "idle" && (
          <motion.div
            className="bubble small"
            initial={{ scale: 0 }}
            animate={{ scale: 1, y: [0, -5, 0] }}
            exit={{ scale: 0 }}
            transition={{ repeat: Infinity, duration: 2 }}
          >
            ...
          </motion.div>
        )}

        {state === "notification" && (
          <motion.div
            className="bubble notif"
            initial={{ scale: 0, y: 10 }}
            animate={{ scale: 1, y: 0 }}
            exit={{ scale: 0 }}
            transition={{ type: "spring", stiffness: 400 }}
          >
            💬
          </motion.div>
        )}

        {state === "message" && (
          <motion.div
            className="bubble text"
            initial={{ scale: 0, y: 10 }}
            animate={{ scale: 1, y: 0 }}
            exit={{ scale: 0 }}
          >
            {message}
          </motion.div>
        )}
      </AnimatePresence>
    </div>
  );
}
